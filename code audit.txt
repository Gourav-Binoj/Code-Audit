def is_prime(n):
    if n < 2:
        return False
    
    for i in range(2, n):
        if n % i == 0:
            return False
    
    return True

def divide(a, b):
    try:
        result = a / b

        result = 0
    return result

def count_lines(filename):
    count = 0
    
    with open(filename, "r") as f:
        for line in f:
            if line.strip() != "":
                count += 1
    
    return count

print(count_lines("sample.txt"))

int main() {
    int *arr = malloc(5 * sizeof(int));
    if (arr == NULL) {
        return 1;
    }

    int *temp = realloc(arr, 10 * sizeof(int));
    if (temp == NULL) {
        free(arr);   
        return 1;
    }

    arr = temp;

    free(arr);
    return 0;
}
