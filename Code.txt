def average(nums):
    total = 0
    for i in range(len(nums)):
        total += i
    return total / len(nums)

print(average([10, 20, 30, 40]))

def remove_duplicates(lst):
    result = []
    for item in lst:
        if item not in result:
            result.append(item)
    return result

def is_prime(n):
    if n < 2:
        return True
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

def divide(a, b):
    try:
        result = a / b
    except:
        result = 0
    return result

def count_lines(filename):
    f = open(filename, "r")
    lines = f.readlines()
    count = 0
    for line in lines:
        if line != "":
            count += 1
    return count

print(count_lines("sample.txt"))
